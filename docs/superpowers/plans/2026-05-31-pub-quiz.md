```js
// ps-offsite-2026/shared/quiz.js
// Pure helpers for the Pub Quiz category/question flow. No Firebase imports
// (so this is unit-testable and usable inside the Firebase-free lobby.js).
//
// Category map (Firebase node lobbies/{id}/quiz/categories):
//   { [catId]: { order:int, name:string, questionCount:int } }
// Submissions for ONE team (lobbies/{id}/quiz/submissions/{teamId}):
//   { [catId]: { submittedAt:int, answers: { [idx]: string } } }

export const DEFAULT_CATEGORY_COUNT = 4;
export const DEFAULT_QUESTION_COUNT = 8;

// Seed map for a new lobby. Deterministic keys c1..cN so it can be created
// inside the Firebase-free createLobby() (which cannot mint push ids).
export function seedCategories(count = DEFAULT_CATEGORY_COUNT, questionCount = DEFAULT_QUESTION_COUNT) {
  const out = {};
  for (let i = 0; i < count; i++) {
    out[`c${i + 1}`] = { order: i, name: `Category ${i + 1}`, questionCount };
  }
  return out;
}

// Categories sorted by order, as [{ id, order, name, questionCount }].
export function orderedCategories(categories) {
  return Object.entries(categories || {})
    .map(([id, c]) => ({
      id,
      order: c?.order ?? 0,
      name: c?.name ?? '',
      questionCount: c?.questionCount ?? 0,
    }))
    .sort((a, b) => a.order - b.order);
}

// First category (by order) the team has NOT submitted, or null if every
// category is submitted / there are no categories.
export function currentCategoryId(categories, submissions) {
  const subs = submissions || {};
  for (const cat of orderedCategories(categories)) {
    if (!subs[cat.id]) return cat.id;
  }
  return null;
}

// True iff there is >=1 category and the team submitted all of them.
export function allCategoriesSubmitted(categories, submissions) {
  return orderedCategories(categories).length > 0
    && currentCategoryId(categories, submissions) === null;
}

// Order int for a newly added category (append to the end).
export function nextOrder(categories) {
  const ordered = orderedCategories(categories);
  return ordered.length ? ordered[ordered.length - 1].order + 1 : 0;
}
```

```html
import { getDatabase, ref, get, set, update, push, onValue } from 'https://www.gstatic.com/firebasejs/10.13.0/firebase-database.js';
import { firebaseConfig } from '../firebase-config.js';
import { resolveSession } from '../shared/lobby.js';
import { isGameLockedFor, renderLockedScreen } from '../shared/game-gate.js';
import { mountTopbar } from '../shared/topbar.js';
import { getGame } from '../shared/games-catalog.js';
import { submitScore, firebaseWriter } from '../shared/score-submit.js';
import { resolveTimer, resolveRule, deadlineFor, isExpired, formatMMSS } from '../shared/game-config.js';
import { orderedCategories, currentCategoryId } from '../shared/quiz.js';

const KEY = 'PQ';
const session = resolveSession();
if (!session) { location.replace('../index.html'); throw new Error('no session'); }
mountTopbar({ activePage: 'games' });

const game = getGame(KEY);
const container = document.getElementById('container');
const catalogHref = `../games.html?lobby=${encodeURIComponent(session.lobbyId)}&team=${session.teamId}`;

function esc(s) {
  return String(s).replace(/[&<>"']/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]));
}
function renderRules(text) {
  const blocks = String(text || '').split(/\n\s*\n/);
  return blocks.map(b => {
    const lines = b.split('\n');
    if (lines.every(l => l.startsWith('- '))) {
      return `<ul>${lines.map(l => `<li>${esc(l.slice(2))}</li>`).join('')}</ul>`;
    }
    return `<p>${esc(b.replace(/\n/g, ' '))}</p>`;
  }).join('');
}

const app = initializeApp(firebaseConfig);
const db = getDatabase(app);
const writer = firebaseWriter({ db, ref, update, push });

const LOBBY = `lobbies/${session.lobbyId}`;
const scorePath = `${LOBBY}/scores/${session.teamId}/${KEY}`;
const startPath = `${LOBBY}/timerStarts/${KEY}/${session.teamId}`;
const subsPath  = `${LOBBY}/quiz/submissions/${session.teamId}`;

// live state
let categories = null;
let catsLoaded = false;
let submissions = null;
let currentRules = null;
let currentScore = null;   // PQ score (null until final submit)
// timer state (mirrors manual.html)
let minutes;
let startTs = null;
let timerId = null;
let settled = false;
let timeUp = false;

function settle() {
  if (settled) return false;
  settled = true;
  if (timerId !== null) { clearInterval(timerId); timerId = null; }
  return true;
}
function rulesText() {
  return resolveRule(currentRules, KEY, session.teamId, game.rules);
}

function shell(inner) {
  return `
    <div class="head"><div class="emoji">${game.emoji}</div><h1>${esc(game.name)}</h1></div>
    <div class="countdown" id="countdown" ${minutes ? '' : 'hidden'}></div>
    <div class="rules">${renderRules(rulesText())}</div>
    ${inner}`;
}

function render() {
  if (timeUp) { renderTimeUp(); return; }
  if (!catsLoaded) return; // keep "Loading…" until first categories snapshot
  const ordered = orderedCategories(categories);
  if (ordered.length === 0) {
    container.innerHTML = shell(`<div class="panel"><h2>Quiz not ready</h2><p class="current">The host hasn't set up any categories yet. Check back soon.</p><a class="back-link" href="${catalogHref}">← Back to catalog</a></div>`);
    if (minutes) paintCountdown();
    return;
  }
  const curId = currentCategoryId(categories, submissions);
  if (curId !== null) { renderCategory(ordered.find(c => c.id === curId), ordered); return; }
  renderScore(ordered);
}

function renderCategory(cat, ordered) {
  const done = ordered.filter(c => (submissions || {})[c.id]).length;
  const n = cat.questionCount || 0;
  let qs = '';
  for (let i = 0; i < n; i++) {
    qs += `<div class="q"><label for="q${i}">Question ${i + 1}</label><input id="q${i}" type="text" autocomplete="off" /></div>`;
  }
  container.innerHTML = shell(`
    <div class="progress">Category ${done + 1} of ${ordered.length}</div>
    <div class="panel">
      <h2>${esc(cat.name)}</h2>
      ${qs}
      <button id="submitCat">Submit category</button>
      <div id="banner"></div>
      <a class="back-link" href="${catalogHref}">← Back to catalog</a>
    </div>`);
  if (minutes) paintCountdown();
  document.getElementById('submitCat').addEventListener('click', () => submitCategory(cat));
}

async function submitCategory(cat) {
  if (timeUp) return;
  const btn = document.getElementById('submitCat');
  const banner = document.getElementById('banner');
  const n = cat.questionCount || 0;
  const answers = {};
  for (let i = 0; i < n; i++) answers[i] = document.getElementById(`q${i}`).value.trim();
  btn.disabled = true;
  try {
    if (await isGameLockedFor({ db, ref, get, lobbyId: session.lobbyId, teamId: session.teamId, gameKey: KEY })) {
      banner.className = 'banner bad'; banner.textContent = 'Locked — not saved.'; btn.disabled = false; return;
    }
    await set(ref(db, `${subsPath}/${cat.id}`), { submittedAt: Date.now(), answers });
    // the live submissions listener re-renders -> next category (or score) appears
  } catch (e) {
    banner.className = 'banner bad'; banner.textContent = 'Save failed: ' + (e.message || 'unknown error');
    btn.disabled = false;
  }
}

function renderScore(ordered) {
  const submitted = currentScore !== null;
  // Timed games are one-shot: after a submit, lock the input (mirrors manual).
  const readOnly = minutes && submitted;
  container.innerHTML = shell(`
    <div class="progress">All ${ordered.length} categories submitted</div>
    <div class="panel">
      ${submitted ? `<div class="current">Currently submitted: <strong>${esc(String(currentScore))}</strong> pts</div>` : ''}
      <label class="score-label" for="score">${submitted ? 'Resubmit raw score' : 'Raw score'}</label>
      <input id="score" type="number" min="0" step="1" inputmode="numeric" value="${submitted ? esc(String(currentScore)) : ''}" ${readOnly ? 'disabled' : ''} />
      <button id="submitScore" ${readOnly ? 'disabled' : ''}>${submitted ? 'Resubmit' : 'Submit score'}</button>
      <div id="banner"></div>
      <a class="back-link" href="${catalogHref}">← Back to catalog</a>
    </div>`);
  if (minutes) paintCountdown();
  const btn = document.getElementById('submitScore');
  if (btn) btn.addEventListener('click', onSubmitScore);
}

async function onSubmitScore() {
  const input = document.getElementById('score');
  const btn = document.getElementById('submitScore');
  const banner = document.getElementById('banner');
  const raw = parseInt(input.value, 10);
  if (!Number.isFinite(raw) || raw < 0) {
    banner.className = 'banner bad'; banner.textContent = 'Enter a non-negative integer.'; return;
  }
  // Timed: settle FIRST (sync, before any await) to win the race vs auto-0.
  if (minutes) { if (!settle()) return; }
  btn.disabled = true; banner.className = ''; banner.textContent = '';
  try {
    if (await isGameLockedFor({ db, ref, get, lobbyId: session.lobbyId, teamId: session.teamId, gameKey: KEY })) {
      banner.className = 'banner bad'; banner.textContent = 'Locked — score not saved.';
      if (minutes) rearmCountdown();
      btn.disabled = false; return;
    }
    const late = minutes && isExpired(startTs, minutes, Date.now());
    const toSave = late ? 0 : raw;
    const saved = await submitScore({ writer, lobbyId: session.lobbyId, teamId: session.teamId, gameKey: KEY, score: toSave });
    banner.className = late ? 'banner bad' : 'banner ok';
    banner.innerHTML = late ? `⏱ Time's up — <strong>0</strong> recorded.` : `Saved <strong>${esc(String(saved))}</strong> pts.`;
    const cd = document.getElementById('countdown'); if (cd) cd.hidden = true;
  } catch (e) {
    banner.className = 'banner bad'; banner.textContent = 'Save failed: ' + (e.message || 'unknown error');
    if (minutes) rearmCountdown();
    btn.disabled = false;
  } finally {
    if (!minutes) btn.disabled = false;
  }
}

async function autoZero() {
  if (!settle()) return;
  timeUp = true;
  try {
    await submitScore({ writer, lobbyId: session.lobbyId, teamId: session.teamId, gameKey: KEY, score: 0 });
  } catch (e) {
    console.error('auto-0 failed', e);
  }
  renderTimeUp();
}

function renderTimeUp() {
  container.innerHTML = shell(`
    <div class="panel">
      <div class="banner bad">⏱ Time's up — <strong>0</strong> recorded.</div>
      <a class="back-link" href="${catalogHref}">← Back to catalog</a>
    </div>`);
  const cd = document.getElementById('countdown');
  if (cd) { cd.hidden = false; cd.classList.add('low'); cd.textContent = "⏱ Time's up"; }
}

function rearmCountdown() {
  settled = false;
  if (isExpired(startTs, minutes, Date.now())) autoZero();
  else startCountdown();
}

function paintCountdown() {
  const cd = document.getElementById('countdown');
  if (!cd || startTs === null) return;
  const remaining = deadlineFor(startTs, minutes) - Date.now();
  cd.hidden = false;
  cd.textContent = `⏱ ${formatMMSS(remaining)} left`;
  cd.classList.toggle('low', remaining <= 30000);
}

function startCountdown() {
  const tick = () => {
    const remaining = deadlineFor(startTs, minutes) - Date.now();
    const cd = document.getElementById('countdown');
    if (cd) {
      cd.hidden = false;
      cd.textContent = `⏱ ${formatMMSS(remaining)} left`;
      cd.classList.toggle('low', remaining <= 30000);
    }
    if (remaining <= 0) autoZero();
  };
  tick();
  timerId = setInterval(tick, 1000);
}

(async () => {
  if (await isGameLockedFor({ db, ref, get, lobbyId: session.lobbyId, teamId: session.teamId, gameKey: KEY })) {
    renderLockedScreen(catalogHref);
    return;
  }
  // Resolve the timer once at entry (mirrors manual.html).
  const timersSnap = await get(ref(db, `${LOBBY}/timers`));
  minutes = resolveTimer(timersSnap.exists() ? timersSnap.val() : null, KEY, session.teamId);
  if (minutes) {
    const startSnap = await get(ref(db, startPath));
    if (startSnap.exists()) {
      startTs = startSnap.val();
    } else {
      startTs = Date.now();
      await set(ref(db, startPath), startTs);
    }
    if (isExpired(startTs, minutes, Date.now())) { timeUp = true; autoZero(); return; }
  }
  // Attach live listeners only AFTER the lock gate passes, so a locked team
  // never briefly sees quiz content.
  onValue(ref(db, `${LOBBY}/quiz/categories`), snap => {
    categories = snap.exists() ? snap.val() : null; catsLoaded = true; render();
  });
  onValue(ref(db, subsPath), snap => {
    submissions = snap.exists() ? snap.val() : null; render();
  });
  onValue(ref(db, scorePath), snap => {
    currentScore = snap.exists() ? snap.val() : null; render();
  });
  onValue(ref(db, `${LOBBY}/rules`), snap => {
    currentRules = snap.exists() ? snap.val() : null;
    const el = container.querySelector('.rules');
    if (el) el.innerHTML = renderRules(rulesText());
  });
  if (minutes) startCountdown();
})();
</script>
</body>
</html>
```
